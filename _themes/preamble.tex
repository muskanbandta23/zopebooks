% ===== COLORS =====
\usepackage{xcolor}
\definecolor{zopdev-indigo}{HTML}{4F46E5}
\definecolor{zopdev-indigo-dark}{HTML}{3730A3}
\definecolor{zopdev-navy}{HTML}{0A1628}
\definecolor{zopdev-gray}{HTML}{525252}
\definecolor{zopdev-light}{HTML}{FAFAFA}
\definecolor{zopdev-border}{HTML}{E5E5E5}
\definecolor{zopdev-success}{HTML}{10B981}
\definecolor{zopdev-warning}{HTML}{F59E0B}
\definecolor{zopdev-danger}{HTML}{EF4444}
% Callout backgrounds — very subtle tints
\definecolor{zopdev-note-bg}{HTML}{F5F3FF}
\definecolor{zopdev-tip-bg}{HTML}{ECFDF5}
\definecolor{zopdev-warning-bg}{HTML}{FFFBEB}
\definecolor{zopdev-important-bg}{HTML}{FEF2F2}

% ===== GEOMETRY =====
\usepackage{geometry}
\geometry{
  paperwidth=8.5in,
  paperheight=11in,
  margin=1in,
  headheight=14pt
}

% ===== TYPOGRAPHY =====
\usepackage{fontspec}
\IfFontExistsTF{Inter}{%
  \setmainfont{Inter}%
}{}
\IfFontExistsTF{JetBrains Mono}{%
  \setmonofont{JetBrains Mono}[Scale=0.85]%
}{%
  \IfFontExistsTF{Menlo}{%
    \setmonofont{Menlo}[Scale=0.85]%
  }{}%
}

% Line spacing
\usepackage{setspace}
\setstretch{1.35}

% ===== HEADERS AND FOOTERS =====
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[L]{\small\textcolor{zopdev-gray}{\leftmark}}
\fancyhead[R]{\small\textcolor{zopdev-gray}{\thepage}}
\fancyfoot[C]{\small\textcolor{zopdev-gray}{zopdev}}
\renewcommand{\headrulewidth}{0.25pt}
\renewcommand{\footrulewidth}{0pt}

% Plain style for chapter opening pages
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[C]{\small\textcolor{zopdev-gray}{\thepage}}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}

% ===== CHAPTER & SECTION FORMATTING =====
\usepackage{titlesec}

% Chapter: semibold navy, generous spacing
\titleformat{\chapter}[display]
  {\normalfont\huge\fontseries{sb}\selectfont\color{zopdev-navy}}
  {\chaptertitlename\ \thechapter}{20pt}{\Huge}
\titlespacing*{\chapter}{0pt}{-20pt}{30pt}

% Section: semibold navy, more top space
\titleformat{\section}
  {\normalfont\Large\fontseries{sb}\selectfont\color{zopdev-navy}}
  {\thesection}{1em}{}
\titlespacing*{\section}{0pt}{2.5em}{1em}

% Subsection: semibold navy
\titleformat{\subsection}
  {\normalfont\large\fontseries{sb}\selectfont\color{zopdev-navy}}
  {\thesubsection}{1em}{}
\titlespacing*{\subsection}{0pt}{1.5em}{0.75em}

% Subsubsection: semibold navy
\titleformat{\subsubsection}
  {\normalfont\normalsize\fontseries{sb}\selectfont\color{zopdev-navy}}
  {\thesubsubsection}{1em}{}
\titlespacing*{\subsubsection}{0pt}{1.25em}{0.5em}

\tcbuselibrary{listings, skins, breakable}

\newtcolorbox{zopdev-note}[1][Note]{

% ===== BLOCKQUOTES =====
% changepage must be loaded with [strict] to avoid option clash with Quarto internals
\PassOptionsToPackage{strict}{changepage}
\usepackage{changepage}
\renewenvironment{quote}{%
  \begin{adjustwidth}{0.5cm}{0cm}%
  \color{zopdev-gray}%
  \noindent\textcolor{zopdev-border}{\vrule width 2pt}\hspace{0.5em}%
  \begin{minipage}{\dimexpr\linewidth-0.5em-2pt}%
}{%
  \end{minipage}%
  \end{adjustwidth}%
  \vspace{0.5em}%
}

% ===== CUSTOM TITLE PAGE =====
\usepackage{tikz}
\makeatletter
\renewcommand{\maketitle}{%
  \begin{titlepage}
    \begin{tikzpicture}[remember picture, overlay]
      % Subtle accent line at top
      \fill[zopdev-indigo] (current page.north west) rectangle ([yshift=-4pt]current page.north east);
    \end{tikzpicture}
    \vspace*{3cm}
    \begin{flushleft}
      {\fontsize{14}{18}\selectfont\color{zopdev-indigo}\fontseries{sb}\selectfont ZOPDEV\par}
      \vspace{2cm}
      {\fontsize{36}{42}\selectfont\fontseries{b}\selectfont\color{zopdev-navy}\@title\par}
      \vspace{0.75cm}
      \ifdefined\@subtitle{\fontsize{16}{22}\selectfont\color{zopdev-gray}\@subtitle\par}\fi
      \vspace{3cm}
      {\fontsize{12}{16}\selectfont\color{zopdev-gray}\@author\par}
      \vspace{0.5cm}
      {\fontsize{11}{14}\selectfont\color{zopdev-border}\@date\par}
    \end{flushleft}
    \vfill
    \begin{flushleft}
      \textcolor{zopdev-border}{\rule{3cm}{0.5pt}}
    \end{flushleft}
  \end{titlepage}
}
\makeatother
